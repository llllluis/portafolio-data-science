SELECT:


SELECT rp.IDALUMNE, rp.IDCURS, rp.IDEVALUACIO, rp.MATERIA, rp.RESULT, rt.TOTAL, ta.POBLACIO, ta.DATANAIXEMENT, ta.SEXO, ta.TIPUS_ESCOLA, tev.DATA, tev.DESCRIPCIO, tc.PROFE, tc.TIPOESTUDIANTS, tee.EJERCICIO_ESCOLAR, tee.INICI
FROM ((((tbRESULTATS_PARCIALS AS rp
INNER JOIN tbRESULTATS_TOTALS AS rt 
    ON rp.IDALUMNE = rt.IDALUMNE 
    AND rp.IDCURS = rt.IDCURS 
    AND rp.IDIOMA = rt.IDIOMA 
    AND rp.IDEVALUACIO = rt.IDEVALUACIO)
INNER JOIN tbALUMNES AS ta 
    ON rp.IDALUMNE = ta.IDALUMNE)
INNER JOIN tbEVALUACIONS AS tev 
    ON rp.IDEVALUACIO = tev.IDEVALUACIO)
INNER JOIN tbCURSOS_2005_2006 AS tc 
    ON rp.IDCURS = tc.IDCURS)
INNER JOIN tbEJERCICIO_ESCOLAR AS tee 
    ON tc.NRO_EJERCICIO = tee.NRO_EJERCICIO
WHERE rp.IDIOMA = 3
  AND tc.NRO_EJERCICIO BETWEEN 10 AND 20
  AND tc.IDTIPOCURS = 1
  AND rt.TOTAL <> 0
  AND rp.RESULT <> 0
